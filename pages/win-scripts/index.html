Windows 脚本

# 清理Maven本地仓库的lastupdated文件

@echo off
set REPOSITORY_PATH=
rem 正在搜索...
for /f "delims=" %%i in ('dir /b /s "%REPOSITORY_PATH%\*lastUpdated*"') do (
    del /s /q %%i
)
rem 搜索完毕
pause

# cd 命令无法切换盘符

# 方法一
cd D:\
# 输入完上面命令后，再输入
D:
# 回车后，路径即成功的切换到我们想要到的路径

# 方法二
cd /dD:\

# 端口占用

# 查看 8080 端口是否被占用
netstat -aon|findstr "8080"

# 查看是那个进程占用 2668 端口
tasklist|findstr "2668"

# 结束进程
taskkill /f /t /im TIM.exe
